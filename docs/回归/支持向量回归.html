支持向量回归#

支持向量分类的方法可以被扩展用作解决回归问题。这个方法被称作支持向量回归。有关支持向量机的内容详见分类部分的介绍。

支持向量回归有三种不同的实现形式: SVR, NuSVR 和 LinearSVR. 在只考虑线性核的情况下, LinearSVR 比 SVR 提供一个更快的实现形式

import numpy as np
import matplotlib.pyplot as plt
from sklearn.svm import SVR

# 生成示例数据
X = np.sort(5 * np.random.rand(100, 1), axis=0)
y = np.sin(X).ravel()

# 创建不同类型的SVR模型
svr_linear = SVR(kernel='linear', C=1.0)
svr_rbf = SVR(kernel='rbf', C=1.0, gamma='scale')
svr_poly = SVR(kernel='poly', C=1.0, degree=5)

# 使用训练数据拟合模型
svr_linear.fit(X, y)
svr_rbf.fit(X, y)
svr_poly.fit(X, y)

# 生成新的测试数据
X_test = np.arange(0.0, 5.0, 0.01)[:, np.newaxis]

# 使用模型进行预测
y_linear = svr_linear.predict(X_test)
y_rbf = svr_rbf.predict(X_test)
y_poly = svr_poly.predict(X_test)

# 可视化结果
plt.scatter(X, y, color='darkorange', label='data')
plt.plot(X_test, y_linear, color='blue', lw=2, label='Linear SVR')
plt.plot(X_test, y_rbf, color='red', lw=2, label='RBF SVR')
plt.plot(X_test, y_poly, color='green', lw=2, label='Poly SVR')
plt.xlabel('X')
plt.ylabel('y')
plt.title('Support Vector Regression')
plt.legend()
plt.show()
